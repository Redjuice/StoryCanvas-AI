```javascript
import { defineStore } from 'pinia'
import { ref } from 'vue'

export const useUserStore = defineStore('user', () => {
  const token = ref(localStorage.getItem('token') || '')
  const userInfo = ref(null)

  const setToken = (newToken) => {
    token.value = newToken
    localStorage.setItem('token', newToken)
  }

  const setUserInfo = (info) => {
    userInfo.value = info
  }

  const logout = () => {
    token.value = ''
    userInfo.value = null
    localStorage.removeItem('token')
  }

  return { token, userInfo, setToken, setUserInfo, logout }
})
```

```vue
            required

        </div>
        <div>
          <label class="block text-sm font-medium text-gray-700">确认密码</label>
          <input
            v-model="form.confirmPassword"
            type="password"
            class="input-field mt-1"
            placeholder="请确认密码"
            required
          />
        </div>
        <button
          type="submit"
          class="btn-primary w-full py-3"
        >
          注册
        </button>
      </form>
      <p class="mt-4 text-center text-sm text-gray-600">
        已有账号？<router-link to="/auth/login" class="text-primary">登录</router-link>
      </p>
    </div>
  </div>
</template>

<script setup>
import { ref } from 'vue'
import { useRouter } from 'vue-router'
import { register } from '@/api/auth'

const router = useRouter()
const form = ref({
  email: '',
  password: '',
  confirmPassword: '',
})

const onRegister = async () => {
  if (form.value.password !== form.value.confirmPassword) {
    alert('两次密码输入不一致')
    return
  }
  try {
    await register(form.value)
    alert('注册成功')
    router.push('/auth/login')
  } catch (error) {
    alert(error.message || '注册失败')
  }
}
</script>
```

### 4.2 故事 API 接口

**src/api/story.js：**

```javascript
import axios from './index'

/**
 * 创建故事
 * @param {Object} data - 故事参数
 * @param {string} data.theme - 故事主题
 * @param {string} data.ageGroup - 目标年龄段
 * @param {string} data.style - 绘画风格
 * @param {number} data.pageCount - 页数
 * @returns {Promise<Object>} 创建的故事
 */
export const createStory = (data) => {
  return axios.post('/stories', data)
}

/**
 * 获取故事详情
 * @param {string} id - 故事ID
 * @returns {Promise<Object>} 故事详情
 */
export const getStoryDetail = (id) => {
  return axios.get(`/stories/${id}`)
}

/**
 * 更新故事
 * @param {string} id - 故事ID
 * @param {Object} data - 更新数据
 * @returns {Promise<Object>} 更新结果
 */
export const updateStory = (id, data) => {
  return axios.patch(`/stories/${id}`, data)
}

/**
 * 删除故事
 * @param {string} id - 故事ID
 * @returns {Promise<void>}
 */
export const deleteStory = (id) => {
  return axios.delete(`/stories/${id}`)
}

/**
 * 获取用户故事列表
 * @param {Object} params - 查询参数
 * @returns {Promise<Array>} 故事列表
 */
export const getStoryList = (params) => {
  return axios.get('/stories', { params })
}

/**
 * 重新生成某一页
 * @param {string} id - 故事ID
 * @param {number} pageNum - 页码
 * @returns {Promise<Object>} 更新结果
 */
export const regeneratePage = (id, pageNum) => {
  return axios.post(`/stories/${id}/regenerate`, { pageNum })
}
```

## 第五阶段：AI生成模块开发

### 5.1 生成状态管理

**src/stores/story.js：**

```javascript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { getStoryDetail } from '@/api/story'

export const useStoryStore = defineStore('story', () => {
  const currentStory = ref(null)
  const generatingPages = ref([])
  const generatingStatus = ref('idle')

  const setCurrentStory = (story) => {
    currentStory.value = story
  }

  const updatePage = (pageIndex, pageData) => {
    if (currentStory.value && currentStory.value.pages[pageIndex]) {
      currentStory.value.pages[pageIndex] = pageData
    }
  }

  const setGeneratingStatus = (status) => {
    generatingStatus.value = status
  }

  const isPageGenerating = (pageIndex) => {
    return generatingPages.value.includes(pageIndex)
  }

  const fetchStoryDetail = async (id) => {
    const story = await getStoryDetail(id)
    setCurrentStory(story)
    return story
  }

  return {
    currentStory,
    generatingPages,
    generatingStatus,
    setCurrentStory,
    updatePage,
    setGeneratingStatus,
    isPageGenerating,
    fetchStoryDetail,
  }
})
```

```vue
        >
          上一页

        <span class="text-gray-600">{{ currentPage + 1 }} / {{ story.pages.length }}</span>
```

### 7.1 书架页面

**功能需求：**

**src/views/bookshelf/index.vue：**

```vue
<template>
  <div class="min-h-screen bg-gray-50">
    <header class="bg-white shadow">
      <div class="max-w-4xl mx-auto px-4 py-4">
        <h1 class="text-xl font-semibold">我的书架</h1>
      </div>
    </header>

    <div class="max-w-4xl mx-auto px-4 py-4">
      <div class="flex space-x-2 mb-6">
        <button
          v-for="tab in tabs"
          :key="tab.value"
          @click="activeTab = tab.value; fetchList()"
          class="px-4 py-2 rounded-lg text-sm transition-colors"
          :class="activeTab === tab.value ? 'bg-primary text-white' : 'bg-white text-gray-600'"
        >
          {{ tab.label }}
        </button>
      </div>

      <div v-if="storyList.length" class="space-y-4">
        <div
          v-for="story in storyList"
          :key="story.id"
          class="bg-white rounded-lg shadow p-4 flex items-center cursor-pointer hover:shadow-md transition-shadow"
          @click="goToPreview(story.id)"
        >
          <div class="w-20 h-20 rounded-lg overflow-hidden flex-shrink-0 bg-gray-200">
            <img :src="story.cover" :alt="story.title" class="w-full h-full object-cover" />
          </div>
          <div class="ml-4 flex-1 min-w-0">
            <h3 class="font-medium text-gray-900 truncate">{{ story.title }}</h3>
            <p class="text-sm text-gray-500 mt-1">
              {{ story.pageCount }}页 · {{ formatDate(story.createdAt) }}
            </p>
          </div>
          <button
            @click.stop="confirmDelete(story)"
            class="ml-4 text-gray-400 hover:text-red-500"
          >
            删除
          </button>
        </div>
      </div>

      <div v-else class="text-center py-12 text-gray-500">
        暂无作品
      </div>
    </div>
  </div>
</template>

<script setup>
import { ref, onMounted } from 'vue'
import { useRouter } from 'vue-router'
import { getStoryList, deleteStory } from '@/api/story'

const router = useRouter()
const tabs = [
  { label: '全部', value: 'all' },
  { label: '已完成', value: 'completed' },
  { label: '生成中', value: 'generating' },
]
const activeTab = ref('all')
const storyList = ref([])

const fetchList = async () => {
  const params = activeTab.value === 'all' ? {} : { status: activeTab.value }
  const res = await getStoryList(params)
  storyList.value = res.data || []
}

const goToPreview = (id) => {
  router.push(`/preview/${id}`)
}

const confirmDelete = async (story) => {
  if (confirm('确认删除此作品？')) {
    await deleteStory(story.id)
    alert('删除成功')
    fetchList()
  }
}

const formatDate = (date) => {
  return new Date(date).toLocaleDateString('zh-CN')
}

onMounted(() => {
  fetchList()
})
</script>
```
